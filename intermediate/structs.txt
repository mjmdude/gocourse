package main

import "fmt"

// / No modification methods
type Person struct {
	firstName string
	lastName  string
	age       int
	address   Address
	PhoneHomeCell
}
type PhoneHomeCell struct {
	home string
	cell string
}

type Address struct {
	city    string
	country string
}

func (p Person) fullName(title string) string {
	return title + " " + p.firstName + " " + p.lastName
}

// Modificaiton methods

func (p *Person) incrementAgeByOne() {
	p.age++
}

func main() {
	p1 := Person{
		firstName: "John",
		lastName:  "Doe",
		age:       30,
		address: Address{
			city:    "London",
			country: "U.K",
		},
		PhoneHomeCell: PhoneHomeCell{
			home: "999999999",
			cell: "888888888",
		},
	}

	p2 := Person{
		firstName: "Jane",
		age:       25,
	}
	p2.address.city = "NY"
	p2.address.country = "USA"
	p3 := Person{
		firstName: "Jane",
		age:       25,
	}
	p3.address.city = "NY"
	p3.address.country = "USA"

	fmt.Println(p1.firstName)
	fmt.Println(p2.firstName)
	fmt.Println(p1.fullName("Mr."))
	fmt.Println(p1.address)
	fmt.Println(p2.address.city)
	fmt.Println(p1.PhoneHomeCell.cell)

	fmt.Println("Are p1 and p2 equal:", p1 == p2)
	fmt.Println("Are p2 and p3 equal:", p3 == p2)

	// Anonymous structs for temp data structs

	user := struct {
		username string
		email    string
	}{
		username: "user123",
		email:    "dude@gmail.com",
	}
	fmt.Println(user.username)
	fmt.Println("Before increment", p1.age)
	p1.incrementAgeByOne()
	fmt.Println("After increment", p1.age)
	// fmt.Println(p1.age)

	// Methods attached to structs
	// Methods can be defined for any type not just structs

}
